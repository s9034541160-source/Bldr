```python
                })
            else:
                result["recommendations"].append(f"–î–æ–∫—É–º–µ–Ω—Ç {doc_info['number']} —É—Å—Ç–∞—Ä–µ–ª, –∑–∞–º–µ–Ω—è—é—â–∏–π –¥–æ–∫—É–º–µ–Ω—Ç –Ω–µ –Ω–∞–π–¥–µ–Ω")
        
        return result
    
    def _extract_document_info(self, file_path: str, content: str = None) -> Dict:
        """–ò–∑–≤–ª–µ—á–µ–Ω–∏–µ –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏ –æ –¥–æ–∫—É–º–µ–Ω—Ç–µ –∏–∑ —Ñ–∞–π–ª–∞ –∏ —Å–æ–¥–µ—Ä–∂–∏–º–æ–≥–æ"""
        info = {"number": "", "title": ""}
        
        # –ü–æ–ø—ã—Ç–∫–∞ –∏–∑–≤–ª–µ—á—å –Ω–æ–º–µ—Ä –∏–∑ –∏–º–µ–Ω–∏ —Ñ–∞–π–ª–∞
        file_name = os.path.basename(file_path).lower()
        norm_patterns = [
            r"(—Å–ø|—Å–Ω–∏–ø|–≥–æ—Å—Ç|–ø–æ—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ|–ø—Ä–∏–∫–∞–∑|—Ñ–∑)\s*[\d\.\-]+",
            r"(—Å–ø|—Å–Ω–∏–ø|–≥–æ—Å—Ç|–ø–æ—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ|–ø—Ä–∏–∫–∞–∑|—Ñ–∑)[\s_]*\d+[\.\-\d]*"
        ]
        
        for pattern in norm_patterns:
            match = re.search(pattern, file_name, re.IGNORECASE)
            if match:
                info["number"] = match.group(0).upper().replace("_", " ")
                break
        
        # –ï—Å–ª–∏ –µ—Å—Ç—å —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ, –∏–∑–≤–ª–µ–∫–∞–µ–º –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –∏–∑ –Ω–µ–≥–æ
        if content:
            # –ü–æ–∏—Å–∫ –Ω–æ–º–µ—Ä–∞ –¥–æ–∫—É–º–µ–Ω—Ç–∞ –≤ —Å–æ–¥–µ—Ä–∂–∏–º–æ–º
            content_patterns = [
                r"(?:–°–ü|–°–ù–∏–ü|–ì–û–°–¢|–ü–æ—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ|–ü—Ä–∏–∫–∞–∑|–§–ó)\s+\d{2,4}[.:]?\d{2,4}[.:]?\d{2,4}[-]?\d{4}",
                r"(?:–°–ü|–°–ù–∏–ü|–ì–û–°–¢|–ü–æ—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ|–ü—Ä–∏–∫–∞–∑|–§–ó)\s+\d{1,4}-[–§—Ñ][–ó–∑]",
                r"(?:–°–ü|–°–ù–∏–ü|–ì–û–°–¢)\s+\d{2}\-\d{2}\-\d{4}",
                r"(?:–°–ü|–°–ù–∏–ü|–ì–û–°–¢)\s+\d{4}\-\d{4}"
            ]
            
            for pattern in content_patterns:
                matches = re.findall(pattern, content, re.IGNORECASE)
                if matches:
                    info["number"] = matches[0].strip()
                    break
            
            # –ü–æ–∏—Å–∫ –Ω–∞–∑–≤–∞–Ω–∏—è –¥–æ–∫—É–º–µ–Ω—Ç–∞
            title_patterns = [
                r"(?:–°–ü|–°–ù–∏–ü|–ì–û–°–¢|–ü–æ—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ|–ü—Ä–∏–∫–∞–∑|–§–ó)\s+[\d\.\-]+\s*[¬´"]([^¬ª"]+)[¬ª"]",
                r"(?:–°–≤–æ–¥\s+–ø—Ä–∞–≤–∏–ª|–°—Ç—Ä–æ–∏—Ç–µ–ª—å–Ω—ã–µ\s+–Ω–æ—Ä–º—ã\s+–∏\s+–ø—Ä–∞–≤–∏–ª–∞|–ì–æ—Å—É–¥–∞—Ä—Å—Ç–≤–µ–Ω–Ω—ã–π\s+—Å—Ç–∞–Ω–¥–∞—Ä—Ç)\s+([^¬ª"]+)",
                r"(?:–°–ü|–°–ù–∏–ü|–ì–û–°–¢)\s+[\d\.\-]+\s*[-‚Äì‚Äî]\s*([^¬ª"]+)"
            ]
            
            for pattern in title_patterns:
                match = re.search(pattern, content, re.IGNORECASE)
                if match:
                    info["title"] = match.group(1).strip()
                    break
        
        return info
    
    def download_actual_document(self, doc_number: str, target_dir: str) -> Optional[str]:
        """–°–∫–∞—á–∏–≤–∞–Ω–∏–µ –∞–∫—Ç—É–∞–ª—å–Ω–æ–π –≤–µ—Ä—Å–∏–∏ –¥–æ–∫—É–º–µ–Ω—Ç–∞"""
        norm_doc = self.normative_db.get_document(doc_number)
        if not norm_doc or not norm_doc.get("pdf_url"):
            return None
        
        try:
            # –°–æ–∑–¥–∞–µ–º –∏–º—è —Ñ–∞–π–ª–∞
            file_name = f"{norm_doc['number']}_{norm_doc['title'].replace(' ', '_')}.pdf"
            file_name = re.sub(r'[\\/*?:"<>|]', "_", file_name)  # –£–¥–∞–ª—è–µ–º –Ω–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã
            file_path = os.path.join(target_dir, file_name)
            
            # –°–∫–∞—á–∏–≤–∞–µ–º —Ñ–∞–π–ª
            response = requests.get(norm_doc["pdf_url"], timeout=30)
            if response.status_code == 200:
                with open(file_path, "wb") as f:
                    f.write(response.content)
                return file_path
            else:
                print(f"–û—à–∏–±–∫–∞ —Å–∫–∞—á–∏–≤–∞–Ω–∏—è {norm_doc['pdf_url']}: {response.status_code}")
                return None
                
        except Exception as e:
            print(f"–û—à–∏–±–∫–∞ –ø—Ä–∏ —Å–∫–∞—á–∏–≤–∞–Ω–∏–∏ –¥–æ–∫—É–º–µ–Ω—Ç–∞ {doc_number}: {e}")
            return None
```

- **–°–∏–º–±–∏–æ–∑ —Å –ø–∞–π–ø–ª–∞–π–Ω–æ–º**: –≠—Ç–∞–ø 0 –≥–æ—Ç–æ–≤–∏—Ç —Ñ–∞–π–ª (–ø–µ—Ä–µ–∏–º–µ–Ω–æ–≤–∞–Ω–Ω—ã–π, –ø—Ä–æ–≤–µ—Ä–µ–Ω–Ω—ã–π) –¥–ª—è –≠—Ç–∞–ø–∞ 1. –í –≠—Ç–∞–ø–µ 4: –£–ª—É—á—à–∏—Ç—å regex –¥–ª—è –ù–¢–î (e.g. –ø–∞—Ç—Ç–µ—Ä–Ω—ã "–°–ü \d+\.\d+"). –í 5: –ü–æ–¥—Å—á–µ—Ç –Ω–æ—Ä–º/–ø—É–Ω–∫—Ç–æ–≤. –í 7: Rubern —Å hints –¥–ª—è –Ω–æ—Ä–º (e.g. "–∏–∑–≤–ª–µ–∫–∏ –ø—É–Ω–∫—Ç—ã –∫–∞–∫ —Ä–∞–±–æ—Ç—ã").

- **–ú–∞—Å—à—Ç–∞–±**: Cron –¥–ª—è –æ–±–Ω–æ–≤–ª–µ–Ω–∏–π (–µ–∂–µ–Ω–µ–¥–µ–ª—å–Ω–æ –ø—Ä–æ–≤–µ—Ä—è—Ç—å –∞–∫—Ç—É–∞–ª—å–Ω–æ—Å—Ç—å –ø–æ –ë–î).

#### –ú–æ–¥–∏—Ñ–∏—Ü–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –ü–∞–π–ø–ª–∞–π–Ω (–° –≠—Ç–∞–ø–æ–º 0 + –ò–∑–º–µ–Ω–µ–Ω–∏—è –¥–ª—è –ù–¢–î)
- **–≠—Ç–∞–ø 0: Pre-Processing –ù–¢–î (–ù–æ–≤—ã–π, –¥–ª—è norms)**  
  **–§–∞–π–ª**: –ù–æ–≤—ã–π `ntd_preprocessor.py` (–≤—ã–∑—ã–≤–∞–µ—Ç—Å—è –ø–µ—Ä–µ–¥ `process_document()` –≤ `unified_rag_trainer.py`, –µ—Å–ª–∏ `doc_type == 'norms'`).  
  **–ß—Ç–æ –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç**: 
  1. **–ó–∞–≥—Ä—É–∑–∫–∞ –∏–∑ –ë–î**: –ë–µ—Ä–µ—Ç entry –∏–∑ JSON (code, pdf_url).
  2. **–°–∫–∞—á–∏–≤–∞–Ω–∏–µ**: `requests.get(pdf_url)` ‚Üí temp_file (e.g. "temp_sp20.pdf").
  3. **–ü—Ä–æ–≤–µ—Ä–∫–∞ –∞–∫—Ç—É–∞–ª—å–Ω–æ—Å—Ç–∏**: Scraping docs.cntd.ru/garantru.ru (BeautifulSoup –Ω–∞ "–¥–µ–π—Å—Ç–≤—É–µ—Ç/–∞–Ω–Ω—É–ª–∏—Ä–æ–≤–∞–Ω"). –ï—Å–ª–∏ —É—Å—Ç–∞—Ä–µ–≤—à–∏–π ‚Äî skip/log "–£—Å—Ç–∞—Ä–µ–≤—à–∏–π: [code]".
  4. **–ü—Ä–æ–≤–µ—Ä–∫–∞ –≤ –ë–î**: SELECT –ø–æ `code` –≤ SQLite (–∏–∑ JSON). –ï—Å–ª–∏ –µ—Å—Ç—å ‚Äî skip.
  5. **–ü–µ—Ä–µ–∏–º–µ–Ω–æ–≤–∞–Ω–∏–µ**: PyMuPDF –ø–∞—Ä—Å–∏—Ç –ø–µ—Ä–≤—É—é —Å—Ç—Ä–∞–Ω–∏—Ü—É (regex –Ω–∞ "–°–ü \d+\.\d+ (–ù–∞–∑–≤–∞–Ω–∏–µ)"); e.g. "sp2309257.pdf" ‚Üí "–°–ü 20.13330.2016 –ù–∞–≥—Ä—É–∑–∫–∏ –∏ –≤–æ–∑–¥–µ–π—Å—Ç–≤–∏—è.pdf".
  6. **–ü–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ**: –í –ø–∞–ø–∫—É –ø–æ `category` (e.g. `/data/ntd/–°—Ç—Ä–æ–∏—Ç–µ–ª—å—Å—Ç–≤–æ/`).
  7. **–õ–æ–≥**: `‚úÖ [Stage 0] –ù–¢–î –æ–±—Ä–∞–±–æ—Ç–∞–Ω: [new_path] (–∞–∫—Ç—É–∞–ª–µ–Ω, —É–Ω–∏–∫–∞–ª–µ–Ω)`.
  **–†–µ–∑—É–ª—å—Ç–∞—Ç**: `processed_path` (–≥–æ—Ç–æ–≤—ã–π —Ñ–∞–π–ª –¥–ª—è –≠—Ç–∞–ø–∞ 1); –µ—Å–ª–∏ fail ‚Äî skip.  
  **–°–≤—è–∑—å**: `processed_path` ‚Üí –≠—Ç–∞–ø 1 (–≤–∞–ª–∏–¥–∞—Ü–∏—è). –ï—Å–ª–∏ –Ω–µ norms ‚Äî –ø—Ä–æ–ø—É—Å–∫ –≠—Ç–∞–ø–∞ 0.  
  **–ü–æ—á–µ–º—É –Ω—É–∂–µ–Ω**: –ê–≤—Ç–æ–º–∞—Ç–∏–∑–∏—Ä—É–µ—Ç —Ä—É—á–Ω–æ–π —Å–∫–∞—á–∏–≤–∞–Ω–∏–µ –¥–ª—è RAG-—Å—Ç–∞—Ä—Ç–∞; –∏–Ω—Ç–µ–≥—Ä–∏—Ä—É–µ—Ç –ë–î. –í—Ä–µ–º—è: ~5‚Äì10 —Å–µ–∫/—Ñ–∞–π–ª.

- **–ò–∑–º–µ–Ω–µ–Ω–∏—è –≤ –°—É—â–µ—Å—Ç–≤—É—é—â–∏—Ö –≠—Ç–∞–ø–∞—Ö (–¢–æ–ª—å–∫–æ –¥–ª—è –ù–¢–î/norms)**:
  - **–≠—Ç–∞–ø 4**: –í `_stage4_document_type_detection`: –î–æ–±–∞–≤—å—Ç–µ NTD-–ø–∞—Ç—Ç–µ—Ä–Ω—ã –≤ regex (e.g. `r'–°–ü \d+\.\d+|–ì–û–°–¢ –† \d+'` ‚Üí type='norms', subtype='sp/gost'). –õ—ë–≥–∫–∏–π Rubern-—Å–∫–∞–Ω: –ü—Ä–æ–≤–µ—Ä—å—Ç–µ –Ω–∞ "–ø. 1.1", "–ü—Ä–∏–ª–æ–∂–µ–Ω–∏–µ –ê". Confidence +0.2 –¥–ª—è NTD. –õ–æ–≥: `(regex: 0.9 NTD-match + rubern: 0.1)`.
  - **–≠—Ç–∞–ø 5**: –í `_stage5_structural_analysis`: –î–ª—è norms ‚Äî –ø–æ–¥—Å—á–µ—Ç "–ø—É–Ω–∫—Ç–æ–≤" (`r'\d+\.\d+'`), "—Ä–∞–∑–¥–µ–ª–æ–≤" (`r'–†–∞–∑–¥–µ–ª \d+'`), "–ø—Ä–∏–ª–æ–∂–µ–Ω–∏–π". Completeness: >20 –ø—É–Ω–∫—Ç–æ–≤ = 0.9. –ò—Å–ø–æ–ª—å–∑—É–π—Ç–µ –¥–ª—è seeds (e.g. "–ø. 5.10" –∫–∞–∫ –∫–∞–Ω–¥–∏–¥–∞—Ç).
  - **–≠—Ç–∞–ø 6**: –í `_stage6_regex_to_rubern`: NTD-–ø–∞—Ç—Ç–µ—Ä–Ω—ã: `r'–ø\.\s*\d+\.\d+|–°–ü \d+\.\d+\s*–ø\.\d+'` (–∏–∑ —Ä–∞–∑–¥–µ–ª–æ–≤ –≠—Ç–∞–ø–∞ 5). Seeds: –ü—É–Ω–∫—Ç—ã –Ω–æ—Ä–º –∫–∞–∫ "—Ä–∞–±–æ—Ç—ã" (e.g. "–ø. 2.3 –ú–æ–Ω—Ç–∞–∂").
  - **–≠—Ç–∞–ø 7**: –í `_stage7_rubern_markup`: Hints –¥–ª—è norms: "–ò–∑–≤–ª–µ–∫–∏ —Å—Ç—Ä—É–∫—Ç—É—Ä—É –∫–∞–∫ –Ω–æ—Ä–º—ã: –ø—É–Ω–∫—Ç—ã –∫–∞–∫ —Ä–∞–±–æ—Ç—ã, –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –ø–æ —Å—Å—ã–ª–∫–∞–º (–ø. 1.2 —Å—Å—ã–ª–∞–µ—Ç—Å—è –Ω–∞ 2.1)". Rubern —Å—Ç—Ä–æ–∏—Ç –≥—Ä–∞—Ñ –Ω–æ—Ä–º (e.g. "–∑–∞–≤–∏—Å–∏–º–æ—Å—Ç—å: –ø.3 ‚Üí –ø.4").
  - **–≠—Ç–∞–ø 8**: –ú–µ—Ç–∞–¥–∞–Ω–Ω—ã–µ –¥–ª—è NTD: –ò–∑ —Å—Ç—Ä—É–∫—Ç—É—Ä—ã 7 ‚Äî "–¥–∞—Ç—ã" (–≥–æ–¥ –≤–≤–µ–¥–µ–Ω–∏—è –∏–∑ –∑–∞–≥–æ–ª–æ–≤–∫–∞), "–º–∞—Ç–µ—Ä–∏–∞–ª—ã" (–∏–∑ –ø—Ä–∏–ª–æ–∂–µ–Ω–∏–π —Å —Ç–∞–±–ª–∏—Ü–∞–º–∏, regex "–±–µ—Ç–æ–Ω –ú300"), "—Ñ–∏–Ω–∞–Ω—Å—ã" (–µ—Å–ª–∏ –≤ —Å–º–µ—Ç–Ω—ã—Ö –Ω–æ—Ä–º–∞—Ö). Doc_numbers: –ò–∑–≤–ª–µ—á—å code/title.
  - **–≠—Ç–∞–ø 9**: Quality: +bonus –∑–∞ NTD-—Å–æ–≤–º–µ—Å—Ç–∏–º–æ—Å—Ç—å (e.g. >10 –ø—É–Ω–∫—Ç–æ–≤ = +20 score).
  - **–≠—Ç–∞–ø 10**: –î–ª—è norms: "–ü—Ä–æ–≤–µ—Ä–∫–∞ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏—è –ø—É–Ω–∫—Ç–æ–≤" (–≤–∞–ª–∏–¥–∞—Ü–∏—è —Å—Å—ã–ª–æ–∫ –≤ 7).
  - **–≠—Ç–∞–ø 13**: –ß–∞–Ω–∫–∏ –ø–æ –ø—É–Ω–∫—Ç–∞–º/—Ä–∞–∑–¥–µ–ª–∞–º (e.g. –∫–∞–∂–¥—ã–π "–ø. X" = —á–∞–Ω–∫ —Å metadata {'ntd_code': code, 'status': status} –∏–∑ –≠—Ç–∞–ø–∞ 0/–ë–î).
  - **–≠—Ç–∞–ø 14**: –í Qdrant: –î–æ–±–∞–≤—å—Ç–µ payload {'category': 'ntd', 'actual': True} –¥–ª—è —Ñ–∏–ª—å—Ç—Ä–∞.

**–û–±—â–∏–µ —É–ª—É—á—à–µ–Ω–∏—è –ø–∞–π–ø–ª–∞–π–Ω–∞**:
- **–í—ã–∑–æ–≤ –≠—Ç–∞–ø–∞ 0**: –í `process_document()`: `if Path(file_path).parent.name == 'ntd': processed_path = ntd_preprocess(file_path, db_conn); file_path = processed_path`.
- **Fallback**: –ï—Å–ª–∏ –≠—Ç–∞–ø 0 fail (—É—Å—Ç–∞—Ä–µ–≤—à–∏–π) ‚Äî mark as "invalid_ntd" –≤ processed_files.json.
- **–õ–æ–≥–∏**: –î–æ–±–∞–≤—å—Ç–µ NTD-—Å–ø–µ—Ü–∏—Ñ–∏–∫—É, e.g. `[Stage 0] –ü–µ—Ä–µ–∏–º–µ–Ω–æ–≤–∞–Ω–æ: sp2309257.pdf ‚Üí –°–ü 20.13330.2016 –ù–∞–≥—Ä—É–∑–∫–∏ –∏ –≤–æ–∑–¥–µ–π—Å—Ç–≤–∏—è.pdf`.
- **–°–∏–º–±–∏–æ–∑ —Å–æ—Ö—Ä–∞–Ω—ë–Ω**: –≠—Ç–∞–ø 0 "–ø–æ–¥–∫–∞—Ä–º–ª–∏–≤–∞–µ—Ç" regex (4-6) —Ç–æ—á–Ω—ã–º–∏ NTD-–ø–∞—Ç—Ç–µ—Ä–Ω–∞–º–∏ –∏–∑ –ë–î; Rubern (7) –∏—Å–ø–æ–ª—å–∑—É–µ—Ç NTD-hints.

#### –û–±–Ω–æ–≤–ª–µ–Ω–Ω—ã–π Python-–ö–æ–¥ (–ò–Ω—Ç–µ–≥—Ä–∞—Ü–∏—è)
**1. `ntd_preprocessor.py` (–≠—Ç–∞–ø 0)**:
```python
import requests
from bs4 import BeautifulSoup
import fitz  # PyMuPDF
import sqlite3
import os
import re
import json
from pathlib import Path

def load_ntd_db(db_path='ntd_full_db.json'):
    with open(db_path, 'r', encoding='utf-8') as f:
        return json.load(f)

def check_actual(code):
    url = f"https://docs.cntd.ru/search?query={code.replace(' ', '%20')}"
    resp = requests.get(url)
    soup = BeautifulSoup(resp.text, 'html.parser')
    status = soup.find(string=re.compile(r'–¥–µ–π—Å—Ç–≤—É–µ—Ç|–∞–Ω–Ω—É–ª–∏—Ä–æ–≤–∞–Ω', re.I))
    return '–¥–µ–π—Å—Ç–≤—É–µ—Ç' in (status.text.lower() if status else '') , status.text if status else '–ù–µ –Ω–∞–π–¥–µ–Ω'

def check_in_local_db(code, conn):
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM ntd WHERE code=?", (code,))
    return cursor.fetchone() is not None

def rename_ntd_file(pdf_path, code, title):
    doc = fitz.open(pdf_path)
    text = doc[0].get_text()  # –ü–µ—Ä–≤–∞—è —Å—Ç—Ä–∞–Ω–∏—Ü–∞
    # Regex –¥–ª—è –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏—è/–∏–∑–≤–ª–µ—á–µ–Ω–∏—è
    match = re.search(r'(–°–ü|–ì–û–°–¢|–°–ù–∏–ü)\s*(\d+\.?\d*[\.\d]*)\s*(.+?)(?=\s*\d|\Z)', text, re.I)
    if match:
        new_title = f"{match.group(1).upper()} {match.group(2)} {match.group(3).strip()}.pdf"
    else:
        new_title = f"{title}.pdf"  # Fallback –∏–∑ –ë–î
    new_path = pdf_path.parent / new_title
    os.rename(pdf_path, new_path)
    return str(new_path)

def ntd_preprocess(code, db_entries, conn, temp_dir="temp_ntd/"):
    entry = next((e for e in db_entries if e['code'] == code), None)
    if not entry:
        return None
    
    url = entry['pdf_url']
    temp_file = Path(temp_dir) / f"{code}.pdf"
    temp_file.parent.mkdir(exist_ok=True)
    
    # –°–∫–∞—á–∏–≤–∞–Ω–∏–µ
    resp = requests.get(url)
    if resp.status_code != 200:
        print(f"‚ùå –°–∫–∞—á–∏–≤–∞–Ω–∏–µ fail: {code}")
        return None
    with open(temp_file, 'wb') as f:
        f.write(resp.content)
    
    # –ê–∫—Ç—É–∞–ª—å–Ω–æ—Å—Ç—å
    is_actual, status_msg = check_actual(code)
    if not is_actual:
        print(f"‚ùå –£—Å—Ç–∞—Ä–µ–≤—à–∏–π: {code} ({status_msg})")
        os.remove(temp_file)
        return None
    
    # –í –ª–æ–∫–∞–ª—å–Ω–æ–π –ë–î
    if check_in_local_db(code, conn):
        print(f"‚ÑπÔ∏è –£–∂–µ –≤ –ë–î: {code}")
        os.remove(temp_file)
        return None
    
    # –ü–µ—Ä–µ–∏–º–µ–Ω–æ–≤–∞–Ω–∏–µ
    renamed = rename_ntd_file(temp_file, code, entry['title'])
    
    # –ü–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ
    category_folder = Path("data/ntd/") / entry['category']
    category_folder.mkdir(exist_ok=True)
    final_path = category_folder / Path(renamed).name
    os.rename(renamed, final_path)
    
    # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –≤ –ª–æ–∫–∞–ª—å–Ω—É—é –ë–î
    cursor = conn.cursor()
    cursor.execute("INSERT INTO ntd (code, title, path, status) VALUES (?, ?, ?, ?)",
                   (code, entry['title'], str(final_path), 'processed'))
    conn.commit()
    
    print(f"‚úÖ [Stage 0] –ù–¢–î: {final_path} (–∞–∫—Ç—É–∞–ª–µ–Ω)")
    return str(final_path)

# –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
conn = sqlite3.connect('ntd_local.db')
conn.execute('''CREATE TABLE IF NOT EXISTS ntd 
                (id INTEGER PRIMARY KEY, code TEXT UNIQUE, title TEXT, path TEXT, status TEXT)''')

# –ü—Ä–∏–º–µ—Ä: –ë–∞—Ç—á –¥–ª—è —Ä—É—á–Ω–æ–≥–æ —Å—Ç–∞—Ä—Ç–∞ (–∏–∑ CSV-–∫–æ–¥–æ–≤)
db_entries = load_ntd_db()
codes_from_csv = ['–°–ü 20.13330.2016', '–°–ü 16.13330.2011']  # –î–æ–±–∞–≤—å—Ç–µ –≤—Å–µ –∏–∑ CSV
for code in codes_from_csv:
    ntd_preprocess(code, db_entries, conn)
```
